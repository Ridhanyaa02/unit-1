```.py
import random
def numbers(a:int,b:int):
    output = a * a + b
    print(output)

input1 = [37,68,60,10]
input2 = [3,2,5,4]
a = random.choice(input1)
b = random.choice(input2)
print(a,b)
numbers(a,b)
```
